## Code
```cpp
class Solution {
public:
    int findUnsortedSubarray(vector<int>& nums) {
        int n = nums.size();
        int i = 0;
        int j = n-1;
        int mini = INT_MAX,maxi = INT_MIN;

        for(int k = 1;k < n;k++) {
            if(nums[k] < nums[k-1])
                mini = min(mini,nums[k]);
        }

        for(int k = 0;k < n-1;k++) {
            if(nums[k] > nums[k+1]) 
                maxi = max(maxi,nums[k]);
        }
        if(mini == INT_MAX && maxi == INT_MIN) return 0;

        while(i < n) {
            if(nums[i] > mini) break;
            i++;
        }

        while(j >= 0) {
            if(nums[j] < maxi) break;
            j--;
        }


       return j-i+1;
    }
};
```
